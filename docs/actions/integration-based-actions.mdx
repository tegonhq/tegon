### Slack Action's config.json

```json
{
  "slug": "slack",
  "name": "Slack",
  "triggers": [
    {
      "type": "on_create",
      "entities": [ "IssueComment", "LinkedIssue" ]
    },
    {
      "type": "source_webhook",
      "entities": [ "slack" ]
    }
  ],
  "integrations": [ "slack" ]
}
```
